Welcome to admob for flash.

admob for flash is a free ane lib for flash developer.it enable flash mobile application load admob advertising. it is easy to use.

user can show all type of banner ad and full screen(interstitial advertising) ad of admob user can show landscope and portrait advertising in actionscript 3 mobile application user can set postion of ad

lib for android can been download from http://code.google.com/p/flash-for-mobile/

Requirements

flash air sdk 3.1 or higher,

iPhone or iPad devices

Recommend flash builder 4.7

useage detail

package
{
    import flash.display.Sprite;
    import so.cuo.anes.admob.AdType;
    import so.cuo.anes.admob.Admob;
    [SWF(width="640", height="960", frameRate="60", backgroundColor="#cccccc")]
    public class AdmobTestSimple extends Sprite
    {
    //a14fde93a405e01  is   admob    app id，login to www.admob.com for more info
        private var gid:String="a14fde93a405e01";
        public function AdmobTestSimple()
        {
            super();
            //create admob instance 
            var ad:Admob=Admob.getInstance();
            //test if this lib work on current runtime
            if(ad.isSupported){
            //set google admob id
                ad.setUnitId(gid);
                //set size of admob,please ref tohttps://developers.google.com/mobile-ads-sdk/docs/admob/intermediate for more info about size
                ad.setAdType(AdType.kGADAdSizeBanner);
                //set the position of ad view.
                ad.setPosition(0,100);
                //set if show ad landscape,if true,ad will show as landscape mode
                ad.setIsLandscape(true);
                //add to displayobject tree ,just as sprite.addChild(displayObject)
                ad.addToDisplayList();
                //load ad
                ad.refresh();
            }else{
                trace("not support");
            }
        }
    }
}
